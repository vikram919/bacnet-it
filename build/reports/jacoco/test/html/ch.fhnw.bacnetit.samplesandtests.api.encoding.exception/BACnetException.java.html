package ch.fhnw.bacnetit.samplesandtests.api.encoding.exception;

public class BACnetException extends Exception {
    private static final long serialVersionUID = -1;

    public BACnetException() {
        super();
    }

    public BACnetException(final String message, final Throwable cause) {
        super(message, cause);
    }

    public BACnetException(final String message) {
        super(message);
    }

    public BACnetException(final Throwable cause) {
        super(cause);
    }
}
