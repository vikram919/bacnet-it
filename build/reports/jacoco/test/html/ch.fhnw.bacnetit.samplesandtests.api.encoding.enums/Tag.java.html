package ch.fhnw.bacnetit.samplesandtests.api.encoding.enums;

public enum Tag {
    OPENING_TAG((byte) 0xe), CLOSING_TAG((byte) 0xf), UNSPECIFIED((byte) 0x0);

    private byte id;

    Tag(final byte id) {
        this.id = id;
    }

    public byte getId() {
        return id;
    }

    public static Tag valueOf(final byte id) {
        if (id == OPENING_TAG.id) {
            return OPENING_TAG;
        }
        if (id == CLOSING_TAG.id) {
            return CLOSING_TAG;
        }

        return UNSPECIFIED;
    }

    public boolean isContextTag() {
        return false;
    }
}
